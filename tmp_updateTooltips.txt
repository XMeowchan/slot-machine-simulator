        // 斡횡엥탬
        function renderCard(card, hidden = false, animate = false) {
            const cardEl = document.createElement('div');
            cardEl.className = `card ${['?', '?'].includes(card.suit) ? 'red' : 'black'}`;
            
            if (hidden) {
                cardEl.className = 'card back';
                cardEl.textContent = '??';
            } else {
                cardEl.innerHTML = `
                    <div class="card-value">${card.value}</div>
                    <div class="card-suit">${card.suit}</div>
                `;
            }

            // 怜瞳劤랙탬珂警속땡뺌
            if (animate) {
                cardEl.style.animation = 'cardDeal 0.3s ease-out';
            }
            
            return cardEl;
        }

        // 뫘劤鞫刻
        function updateDisplay(animateNew = false) {
            // 鏤소돨탬
            const dealerCardsEl = document.getElementById('dealerCards');
            const currentDealerCards = Array.from(dealerCardsEl.children);
            
            // 怜警속劤탬
            for (let i = currentDealerCards.length; i < dealerHand.length; i++) {
                const card = dealerHand[i];
                const hidden = gameInProgress && i === 0 && dealerHiddenCard;
                const cardEl = renderCard(card, hidden, animateNew);
                dealerCardsEl.appendChild(cardEl);
            }

            // 흔벎矜狼럇역鏤소돨뒤寧蠟탬
            if (!dealerHiddenCard && currentDealerCards.length > 0) {
                const firstCard = currentDealerCards[0];
                if (firstCard.classList.contains('back')) {
                    const newCard = renderCard(dealerHand[0], false, false);
                    dealerCardsEl.replaceChild(newCard, firstCard);
                }
            }
